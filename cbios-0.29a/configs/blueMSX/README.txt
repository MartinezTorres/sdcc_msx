C-BIOS machine configurations for blueMSX
=========================================

Installation instructions:
1. Copy "MSX - C-BIOS", "MSX2 - C-BIOS", "MSX2+ - C-BIOS",
   "MSX - C-BIOS - JP", "MSX2 - C-BIOS - JP" and "MSX2+ - C-BIOS - JP",
   "MSX - C-BIOS - BR", "MSX2 - C-BIOS - BR" and "MSX2+ - C-BIOS - BR"
   "MSX - C-BIOS - EU", "MSX2 - C-BIOS - EU" and "MSX2+ - C-BIOS - EU",
   subdirectories to "Machines" in the blueMSX installation directory.
2. Copy "cbios_main_msx1.rom" and "cbios_logo_msx1.rom" into the
   "MSX - C-BIOS" directory.
3. Copy "cbios_main_msx1_jp.rom" and "cbios_logo_msx1.rom" into the
   "MSX - C-BIOS - JP" directory.
4. Copy "cbios_main_msx1_br.rom" and "cbios_logo_msx1.rom" into the
   "MSX - C-BIOS - BR" directory.
5. Copy "cbios_main_msx1_eu.rom" and "cbios_logo_msx1.rom" into the
   "MSX - C-BIOS - EU" directory.
6. Copy "cbios_main_msx2.rom", "cbios_logo_msx2.rom" and "cbios_sub.rom" into
   the "MSX2 - C-BIOS" directory.
7. Copy "cbios_main_msx2_jp.rom", "cbios_logo_msx2.rom" and "cbios_sub.rom" into
   the "MSX2 - C-BIOS - JP" directory.
8. Copy "cbios_main_msx2_br.rom", "cbios_logo_msx2.rom" and "cbios_sub.rom" into
   the "MSX2 - C-BIOS - BR" directory.
9. Copy "cbios_main_msx2_eu.rom", "cbios_logo_msx2.rom" and "cbios_sub.rom" into
   the "MSX2 - C-BIOS - EU" directory.
10. Copy "cbios_main_msx2+.rom", "cbios_logo_msx2+.rom", "cbios_sub.rom" and
   "cbios_music.rom" into the "MSX2+ - C-BIOS" directory.
11. Copy "cbios_main_msx2+_jp.rom", "cbios_logo_msx2+.rom", "cbios_sub.rom" and
    "cbios_music.rom" into the "MSX2+ - C-BIOS - JP" directory.
12. Copy "cbios_main_msx2+_br.rom", "cbios_logo_msx2+.rom", "cbios_sub.rom" and
    "cbios_music.rom" into the "MSX2+ - C-BIOS - BR" directory.
13. Copy "cbios_main_msx2+_eu.rom", "cbios_logo_msx2+.rom", "cbios_sub.rom" and
    "cbios_music.rom" into the "MSX2+ - C-BIOS - EU" directory.
